# TOTP (Double authentification) — Plan d'implémentation

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ajouter une double authentification TOTP optionnelle (Google Authenticator / Authy) pour le compte ADMIN, activable/désactivable depuis les Paramètres.

**Architecture:** Le secret TOTP est généré à la demande (`/api/auth/totp/setup`), puis activé après vérification du premier code (`/api/auth/totp/enable`). Le flux de login NextAuth est étendu pour détecter si le TOTP est requis et demander le code en étape 2.

**Tech Stack:** `otplib` (génération + vérification TOTP RFC 6238), `qrcode` (QR code base64), NextAuth v5 credentials, Prisma 5, React 19, Vitest

Cette route génère un nouveau secret TOTP et retourne le QR code. Elle ne stocke **rien** en DB — l'activation se fait dans la route `enable`.

- [ ] **Step 1 : Créer le fichier**

```typescript
// src/app/api/auth/totp/setup/route.ts
import { auth } from "@/lib/auth";
import { authenticator } from "otplib";
import qrcode from "qrcode";
import { NextResponse } from "next/server";

export async function POST() {
  try {
    const session = await auth();
    if (!session?.user?.id) {
      return NextResponse.json({ error: "Non autorisé" }, { status: 401 });
    }
    if (session.user.role !== "ADMIN" && session.user.role !== "SUPER_ADMIN") {
      return NextResponse.json({ error: "Accès non autorisé" }, { status: 403 });
    }

    const secret = authenticator.generateSecret();
    const email = session.user.email ?? "user";
    const uri = authenticator.keyuri(email, "CertPilot", secret);
    const qrCode = await qrcode.toDataURL(uri);

    return NextResponse.json({ secret, qrCode });
  } catch (error) {
    console.error("POST totp/setup error:", error);
    return NextResponse.json({ error: "Erreur serveur" }, { status: 500 });
  }
}
```

```typescript
// src/app/api/auth/totp/enable/route.ts
import { auth } from "@/lib/auth";
import { auditUpdate } from "@/lib/audit";
import prisma from "@/lib/prisma";
import { authenticator } from "otplib";
import { NextRequest, NextResponse } from "next/server";

export async function POST(request: NextRequest) {
  try {
    const session = await auth();
    if (!session?.user?.id) {
      return NextResponse.json({ error: "Non autorisé" }, { status: 401 });
    }
    if (session.user.role !== "ADMIN" && session.user.role !== "SUPER_ADMIN") {
      return NextResponse.json({ error: "Accès non autorisé" }, { status: 403 });
    }

    const body = await request.json();
    const { secret, code } = body as { secret: string; code: string };

    if (!secret || !code) {
      return NextResponse.json({ error: "Secret et code requis" }, { status: 400 });
    }

    const isValid = authenticator.verify({ token: code.trim(), secret });
    if (!isValid) {
      return NextResponse.json({ error: "Code incorrect" }, { status: 400 });
    }

    await prisma.user.update({
      where: { id: session.user.id },
      data: { totpSecret: secret, totpEnabled: true },
    });

    await auditUpdate(
      "USER",
      session.user.id,
      session.user.name ?? session.user.email ?? "Utilisateur",
      { totpEnabled: false },
      { totpEnabled: true },
      {
        id: session.user.id,
        name: session.user.name ?? undefined,
        email: session.user.email ?? undefined,
        companyId: session.user.companyId,
      },
    );

    return NextResponse.json({ success: true });
  } catch (error) {
    console.error("POST totp/enable error:", error);
    return NextResponse.json({ error: "Erreur serveur" }, { status: 500 });
  }
}
```

```typescript
// src/app/api/auth/totp/disable/route.ts
import { auth } from "@/lib/auth";
import { auditUpdate } from "@/lib/audit";
import prisma from "@/lib/prisma";
import { authenticator } from "otplib";
import { NextRequest, NextResponse } from "next/server";

export async function POST(request: NextRequest) {
  try {
    const session = await auth();
    if (!session?.user?.id) {
      return NextResponse.json({ error: "Non autorisé" }, { status: 401 });
    }
    if (session.user.role !== "ADMIN" && session.user.role !== "SUPER_ADMIN") {
      return NextResponse.json({ error: "Accès non autorisé" }, { status: 403 });
    }

    const body = await request.json();
    const { code } = body as { code: string };

    if (!code) {
      return NextResponse.json({ error: "Code requis" }, { status: 400 });
    }

    const user = await prisma.user.findUnique({
      where: { id: session.user.id },
      select: { totpSecret: true, totpEnabled: true },
    });

    if (!user?.totpEnabled || !user.totpSecret) {
      return NextResponse.json({ error: "TOTP non activé" }, { status: 400 });
    }

    const isValid = authenticator.verify({ token: code.trim(), secret: user.totpSecret });
    if (!isValid) {
      return NextResponse.json({ error: "Code incorrect" }, { status: 400 });
    }

    await prisma.user.update({
      where: { id: session.user.id },
      data: { totpSecret: null, totpEnabled: false },
    });

    await auditUpdate(
      "USER",
      session.user.id,
      session.user.name ?? session.user.email ?? "Utilisateur",
      { totpEnabled: true },
      { totpEnabled: false },
      {
        id: session.user.id,
        name: session.user.name ?? undefined,
        email: session.user.email ?? undefined,
        companyId: session.user.companyId,
      },
    );

    return NextResponse.json({ success: true });
  } catch (error) {
    console.error("POST totp/disable error:", error);
    return NextResponse.json({ error: "Erreur serveur" }, { status: 500 });
  }
}
```

```typescript
// src/app/dashboard/settings/totp-settings.tsx
"use client";

import { Button } from "@/components/ui/button";
import { CheckCircle, Loader2, ShieldCheck, ShieldOff } from "lucide-react";
import Image from "next/image";
import { useState } from "react";

interface TotpSettingsProps {
  totpEnabled: boolean;
}

export function TotpSettings({ totpEnabled: initialEnabled }: TotpSettingsProps) {
  const [enabled, setEnabled] = useState(initialEnabled);

  // États du flow d'activation
  const [setupData, setSetupData] = useState<{ secret: string; qrCode: string } | null>(null);
  const [enableCode, setEnableCode] = useState("");
  const [enableLoading, setEnableLoading] = useState(false);
  const [enableError, setEnableError] = useState("");

  // États du flow de désactivation
  const [showDisableForm, setShowDisableForm] = useState(false);
  const [disableCode, setDisableCode] = useState("");
  const [disableLoading, setDisableLoading] = useState(false);
  const [disableError, setDisableError] = useState("");

  const [setupLoading, setSetupLoading] = useState(false);

  async function handleStartSetup() {
    setSetupLoading(true);
    setEnableError("");
    try {
      const res = await fetch("/api/auth/totp/setup", { method: "POST" });
      const data = await res.json();
      if (res.ok) {
        setSetupData(data);
        setEnableCode("");
      } else {
        setEnableError(data.error ?? "Erreur lors de la génération");
      }
    } finally {
      setSetupLoading(false);
    }
  }

  async function handleEnable(e: React.FormEvent) {
    e.preventDefault();
    if (!setupData) return;
    setEnableLoading(true);
    setEnableError("");
    try {
      const res = await fetch("/api/auth/totp/enable", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ secret: setupData.secret, code: enableCode }),
      });
      const data = await res.json();
      if (res.ok) {
        setEnabled(true);
        setSetupData(null);
        setEnableCode("");
      } else {
        setEnableError(data.error ?? "Code incorrect");
      }
    } finally {
      setEnableLoading(false);
    }
  }

  async function handleDisable(e: React.FormEvent) {
    e.preventDefault();
    setDisableLoading(true);
    setDisableError("");
    try {
      const res = await fetch("/api/auth/totp/disable", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ code: disableCode }),
      });
      const data = await res.json();
      if (res.ok) {
        setEnabled(false);
        setShowDisableForm(false);
        setDisableCode("");
      } else {
        setDisableError(data.error ?? "Code incorrect");
      }
    } finally {
      setDisableLoading(false);
    }
  }

  // TOTP activé
  if (enabled) {
    return (
      <div className="space-y-4">
        <div className="flex items-center gap-3 rounded-lg border border-emerald-200 bg-emerald-50 px-4 py-3">
          <CheckCircle className="h-5 w-5 text-emerald-600 shrink-0" />
          <div>
            <p className="text-sm font-medium text-emerald-800">
              Double authentification activée
            </p>
            <p className="text-xs text-emerald-600">
              Votre compte est protégé par un code TOTP à chaque connexion.
            </p>
          </div>
        </div>

        {!showDisableForm ? (
          <Button
            variant="outline"
            size="sm"
            className="text-red-600 hover:text-red-700 hover:border-red-300"
            onClick={() => setShowDisableForm(true)}
          >
            <ShieldOff className="mr-2 h-4 w-4" />
            Désactiver
          </Button>
        ) : (
          <form onSubmit={handleDisable} className="space-y-3 rounded-lg border border-red-100 bg-red-50 p-4">
            <p className="text-sm font-medium text-red-800">
              Confirmez avec votre code TOTP actuel
            </p>
            <input
              type="text"
              inputMode="numeric"
              maxLength={6}
              value={disableCode}
              onChange={(e) => setDisableCode(e.target.value.replace(/\D/g, ""))}
              placeholder="000000"
              autoFocus
              className="w-40 rounded-lg border border-red-200 px-3 py-2 text-center font-mono tracking-widest text-sm focus:border-red-400 focus:outline-none"
            />
            {disableError && (
              <p className="text-xs text-red-600">{disableError}</p>
            )}
            <div className="flex gap-2">
              <Button type="submit" variant="destructive" size="sm" disabled={disableLoading || disableCode.length !== 6}>
                {disableLoading ? <Loader2 className="mr-2 h-3 w-3 animate-spin" /> : null}
                Confirmer la désactivation
              </Button>
              <Button
                type="button"
                variant="outline"
                size="sm"
                onClick={() => { setShowDisableForm(false); setDisableCode(""); setDisableError(""); }}
              >
                Annuler
              </Button>
            </div>
          </form>
        )}
      </div>
    );
  }

  // TOTP désactivé — pas encore de setup en cours
  if (!setupData) {
    return (
      <div className="space-y-4">
        <div className="flex items-center gap-3 rounded-lg border border-gray-200 bg-gray-50 px-4 py-3">
          <ShieldCheck className="h-5 w-5 text-gray-400 shrink-0" />
          <div>
            <p className="text-sm font-medium text-gray-700">
              Double authentification désactivée
            </p>
            <p className="text-xs text-gray-500">
              Activez le TOTP pour sécuriser votre compte avec Google Authenticator ou Authy.
            </p>
          </div>
        </div>
        <Button
          variant="outline"
          size="sm"
          onClick={handleStartSetup}
          disabled={setupLoading}
        >
          {setupLoading ? <Loader2 className="mr-2 h-4 w-4 animate-spin" /> : <ShieldCheck className="mr-2 h-4 w-4" />}
          Activer la double authentification
        </Button>
      </div>
    );
  }

  // TOTP désactivé — setup en cours (QR code affiché)
  return (
    <div className="space-y-4">
      <p className="text-sm text-gray-700 font-medium">
        1. Scannez ce QR code avec Google Authenticator ou Authy

      <div className="flex justify-center">
        <Image src={setupData.qrCode} alt="QR Code TOTP" width={180} height={180} unoptimized />
      </div>
      <p className="text-xs text-gray-500 text-center">
        Ou saisissez manuellement le secret :{" "}
        <code className="rounded bg-gray-100 px-1 py-0.5 font-mono text-xs text-gray-800">
          {setupData.secret}
        </code>
      </p>

      <form onSubmit={handleEnable} className="space-y-3">
        <p className="text-sm text-gray-700 font-medium">
          2. Entrez le code affiché par l&apos;application pour confirmer
        </p>
        <input
          type="text"
          inputMode="numeric"
          maxLength={6}
          value={enableCode}
          onChange={(e) => setEnableCode(e.target.value.replace(/\D/g, ""))}
          placeholder="000000"
          autoFocus
          className="w-40 rounded-lg border border-gray-300 px-3 py-2 text-center font-mono tracking-widest text-sm focus:border-emerald-500 focus:outline-none focus:ring-2 focus:ring-emerald-500/20"
        />
        {enableError && <p className="text-xs text-red-600">{enableError}</p>}
        <div className="flex gap-2">

            {enableLoading ? <Loader2 className="mr-2 h-3 w-3 animate-spin" /> : null}
            Activer
          </Button>
          <Button
            type="button"
            variant="outline"
            size="sm"
            onClick={() => { setSetupData(null); setEnableCode(""); setEnableError(""); }}
          >
            Annuler
          </Button>
        </div>
      </form>
    </div>
  );
```
